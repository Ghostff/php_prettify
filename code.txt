<?php

namespace Text;

class Text extends TextProcess implements Php
{
    /**
     * An array of the types that have been resolved.
     *
     * @var array
     */
    protected $resolved = array();

    public function isShared(string $abstract, int $counter): bool
    {
        self::init(mt_rand(1, 55), parent::__construct());
        if (isset($this->bindings[$abstract]['shared']))
        {
            #cast to bool
            (bool) $shared = $this->bindings[$abstract]['shared'];
        }
        else
        {
            $shared = false;
        }

        return isset($this->instances[$abstract]) || $shared === true;
    }
}
